## Button variants

export const variants = buttonVariants

export const VariantsTemplate = (args) => ({
  template: `
<div class="flex gap-4 flex-wrap">
<VButton v-for="variant in buttonVariants"
    :variant="variant"
    :key="variant"
    :v-bind="args">
    {{ capitalize(variant) }}
    </VButton>
  </div>`,
  components: { VButton },
  data: () => ({
    buttonVariants,
  }),
  methods: {
    capitalize(str) {
      return capital(str)
    },
  },
  setup() {
    return { args }
  },
})
